import React, {Component} from 'react'
import {Container, Row, Col, ListGroup, ListGroupItem, CardImg, Card, CardBody, CardText, CardTitle} from 'reactstrap'
import Pane from '../Pane';
import tower from './tower.jpg'
import coorsSunset from './coorsSunset.jpg'
import firstNotch from './firstNotch.jpg'
import HuntingtonBeach from './HuntingtonBeach.jpg'
 
/* About Us
 */
export default class AboutUs extends Component{
  constructor(props) {
    super(props);
      this.state = {
          team: [
              {name: 'Garrett St. Amand', description: 'I am a Computer Science senior at CSU. I like board games, video games, hiking, and swimming.', image: tower},
              {name: 'Kyle Pashak', description: 'I am a Junior at CSU working towards a Computer Science degree. In my opinion baseball and Dota 2 are the greatest games on the planet and I also enjoy going to the gym.', image: coorsSunset},
              {name: 'Evan Counihan', description: 'I am a Computer Science student at CSU from Breckenridge, CO. My main interests include skiing, hiking, and video games', image: firstNotch},
              {name: 'Patrick Lee', description: 'I am a junior at CSU with a major in Computer Science, and minors in Math and Philosophy. I am from Sterling, CO, but I like California a whole lot more.', image: HuntingtonBeach}
          ]
      }
  }
 
  render() {
    return(
        <Container>
          <Row>
            <Col xs="12">
              {this.heading()}
            </Col>
          </Row>
            {this.CreateAboutUsEntries()}
        </Container>
    )
  }
 
 
  heading() {
    return (
        <Pane header={'About Us'}
              bodyJSX={'This is information about us'}/>
    );
  }
  CreateAboutUsEntries(){
      return(
      <ListGroup>
          {
              this.state.team.map( teamMember =>
                  <ListGroupItem name={teamMember.name}>
                      <Card>
                          <CardImg top width="100%" src={teamMember.image} alt="Card image cap" />
                          <CardBody>
                              <CardTitle><h4>{teamMember.name}</h4></CardTitle>
                              <CardText>{teamMember.description}</CardText>
                          </CardBody>
                      </Card>
                  </ListGroupItem>
              )
          }
      </ListGroup>
      )
  }
 
}

export default class Itinerary extends Component{
    constructor(props) {
        super(props);
 
        this.addLocation = this.addLocation.bind(this);
        this.onChange = this.onChange.bind(this);
        this.sendItinerary = this.sendItinerary.bind(this);

        this.removeLocation = this.removeLocation.bind(this);

        this.downloadItinerary = this.downloadItinerary.bind(this);
        this.itineraryToJson = this.itineraryToJson.bind(this);
        this.moveLocation = this.moveLocation.bind(this);
        this.createFindRequest = this.createFindRequest.bind(this);
        this.sendFindRequest = this.sendFindRequest.bind(this);

    }
 
    render() {
        return(
            <Container>
                <Form>
                    <Row>
                        <Col md={3}>

                                          <Input name="inputDestinationName" type="text"

                                          <Input name="limit" type="text" value={this.props.limit} onChange={this.onChange} id="nameFields" />

                    {this.displayItinerary()}

    heading() {
        return (
            <Pane header={'Itinerary'}
                  bodyJSX={'This is your itinerary'}/>
        );
    }
 
 
    listFoundLocations(){
        if(this.props.itineraryFindPlaces.length > 0) {
            return (
                <Pane header={this.props.foundPlaces + " Places Found with a Limit of " + this.props.limit}
                      bodyJSX={
                          <ListGroup>
                              {
                                  this.props.itineraryFindPlaces.map(itineraryFindListItem =>
                                      <ListGroupItem>
                                          <ItineraryFindListItem itineraryFindListItem={itineraryFindListItem}
                                                                 addLocation={this.addFindLocation}/>
                                      </ListGroupItem>
                                  )
                              }
                          </ListGroup>
                      }/>);
        }
    }
 
    addFindLocation(itineraryFindListItem){
        this.props.addFindLocation(itineraryFindListItem);
    }
 
 
 
    createFindRequest() {
        return{
            'requestType': 'find',
            'requestVersion': 3,
            'match': this.props.match,
            'limit': this.props.limit,
            'found': this.props.foundPlaces,
            'places': this.props.itineraryFindPlaces
        };
    }
 
    sendFindRequest() {
        const request = this.createFindRequest();
        sendServerRequestWithBody('find', request, this.props.settings.serverPort)
            .then((response) => {
                if (response.statusCode >= 200 && response.statusCode <= 299) {
                    this.updatePlacesFind(response.body.places);
                    this.updateFoundFind(response.body.found);
                    this.setState({errorMessage: null});
                }
                else {
                    this.setState({
                        errorMessage: this.props.createErrorBanner(
                            response.statusText,
                            response.statusCode,
                            `Request to ${ this.props.settings.serverPort } failed.`
                        )
                    });
                }
            });
    }
 
 
    updatePlacesFind(value){
        this.props.updateFindPlaces(value);
    }
 
    updateFoundFind(value){
        this.props.updateFindFound(value);
    }
 
 
 
    readFile(files){
        let fileReader = new FileReader();
        fileReader.onloadend = this.handleFileRead;
        fileReader.readAsText(files[0]);
    }
 
    handleFileRead(event){
        this.props.handleFileRead(event);
    }
 
    onChange(event){
        this.props.onChange(event.target.name, event.target.value);
    };
 
    addLocation(){
        this.props.addLocation();
    };
    itineraryToJson(){
        return {
            //let tripTitle = {'title' : this.props.inputTripName};
            'requestType'        : 'itinerary',
            'requestVersion'     : 3,
            'options'      : {'title' : this.props.options.title, 'earthRadius' : this.props.options.units[this.props.options.activeUnit], 'optimization' : this.props.activeOptimization},
            'places' : this.props.itineraryPlaces,
            'distances' : this.props.itineraryDistances
        };
 
 
    }
    sendItinerary() {
        const tipConfigRequest = this.itineraryToJson();
        sendServerRequestWithBody('itinerary', tipConfigRequest, this.props.settings.serverPort)
            .then((response) => {
                if(response.statusCode >= 200 && response.statusCode <= 299) {
                    this.updateDistances(response.body.distances);
                    this.props.updatePlaces(response.body.places);
                    this.setState({errorMessage: null});
                }
                else {
                    this.setState({
                        errorMessage: this.props.createErrorBanner(
                            response.statusText,
                            response.statusCode,
                            `Request to ${ this.props.settings.serverPort } failed.`
                        )
                    });
                }
            });
    }
    displayItinerary(){
        if(this.locationListReady()){
            return (
                <ListGroup>
                    {
                        this.props.itineraryPlaces.map(itineraryListItem =>
                            <ListGroupItem>
                                <ItineraryListItem itineraryListItem={itineraryListItem}
                                                   removeLocation={this.removeLocation}
                                                   moveLocation={this.moveLocation}
                                />
                            </ListGroupItem>
                        )
                    }
                </ListGroup>
            )
        }
    }
 
    removeLocation(itineraryListItem){
        this.props.removeLocation(itineraryListItem);
    }
    moveLocation(itineraryListItem, direction){
        this.props.moveLocation(itineraryListItem, direction);
    }
    updateDistances(distances){
        this.props.updateDistances(distances);
    }
    renderMap() {
        return (
            <Pane header={'Itinerary Map'}
                  bodyJSX={this.renderLeafletMap()}/>
        );
    }
 
    renderLeafletMap() {
        // initial map placement can use either of these approaches:
        // 1: bounds={this.coloradoGeographicBoundaries()}
        // 2: center={this.csuOvalGeographicCoordinates()} zoom={10}
        return (
            <Map bounds={this.coloradoGeographicBoundaries()}
                 style={{height: 500, maxwidth: 700}}>
                <TileLayer url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
                           attribution="&copy; <a href=&quot;http://osm.org/copyright&quot;>OpenStreetMap</a> contributors"
                />
                {this.props.itineraryPlaces.map( itineraryListItem =>
                    <Marker position={(this.locationListReady())? L.latLng(itineraryListItem.latitude, itineraryListItem.longitude): this.csuOvalGeographicCoordinates()}
                            icon={this.markerIcon()}>
                        <Popup className="font-weight-extrabold">{itineraryListItem.name}</Popup>
                    </Marker>
                )}
                /*{this.createLatLonDistList().map( latLongDistPair =>
                    <Polyline
                        positions={[latLongDistPair[0]]}
                        color = 'blue'>
                        <Popup className="font-weight-extrabold">{(this.distanceListReady())? latLongDistPair[1] + " " + this.props.options.activeUnit: 'calculating'}</Popup>
                    </Polyline>
                )}*/
            </Map>
        )
    }
    downloadItinerary(){
 
        const tipConfigRequest = this.itineraryToJson();
 
        let json = JSON.stringify(tipConfigRequest);
        let blob = new Blob([json], {type: "octet/stream"});
        let url = window.URL.createObjectURL(blob);
        window.location.assign(url);
    };
 
 
    markerIcon() {
        // react-leaflet does not currently handle default marker icons correctly,
        // so we must create our own
        return L.icon({
            iconUrl: icon,
            shadowUrl: iconShadow,
            iconAnchor: [12,40]  // for proper placement
        })
    }
 
    locationListReady(){
        return !(this.props.itineraryPlaces === undefined || this.props.itineraryPlaces.length === 0);
    }
 
    distanceListReady(){
        return !(this.props.itineraryDistances === undefined || this.props.itineraryDistances === 0);
    }
 
 
    createLatLonDistList(){
        if(!this.locationListReady() || !this.distanceListReady()){
            return this.csuOvalGeographicCoordinates();
        }
        let latlongs = [];
        for(let i = 0; i < this.props.itineraryPlaces.length-1; i++){
            let pair = [];
            let halfpair1 = [];
            let halfpair2 = [];
            halfpair1.push(this.props.itineraryPlaces[i].latitude);
            halfpair1.push(this.props.itineraryPlaces[i].longitude);
            halfpair2.push(this.props.itineraryPlaces[i+1].latitude);
            halfpair2.push(this.props.itineraryPlaces[i+1].longitude);
            pair.push(halfpair1);
            pair.push(halfpair2);
            latlongs.push([pair, this.props.itineraryDistances[i]]);
        }
        let lastpair = [];
        let firstloc = [];
        let lastloc = [];
        firstloc.push(this.props.itineraryPlaces[0].latitude);
        firstloc.push(this.props.itineraryPlaces[0].longitude);
        lastloc.push(this.props.itineraryPlaces[this.props.itineraryPlaces.length-1].latitude);
        lastloc.push(this.props.itineraryPlaces[this.props.itineraryPlaces.length-1].longitude);
        lastpair.push(lastloc);
        lastpair.push(firstloc);
        latlongs.push([lastpair, this.props.itineraryDistances[this.props.itineraryDistances.length-1]]);
 
        return latlongs;
    }
 
    createTableEntryList(){
        if(!this.locationListReady() || !this.distanceListReady()){
            return [];
        }
        let tableEntryList = [];
        let totDist = 0;
        for(let i = 0; i < this.props.itineraryPlaces.length; i++){
            let tableEntry = [];
            tableEntry.push(i+1);
            tableEntry.push(this.props.itineraryPlaces[i].name);
            tableEntry.push(this.props.itineraryPlaces[i].latitude);
            tableEntry.push(this.props.itineraryPlaces[i].longitude);
            tableEntry.push(this.props.itineraryDistances[i]);
            totDist = totDist + this.props.itineraryDistances[i];
            tableEntry.push(totDist);
            tableEntryList.push(tableEntry);
        }
        return tableEntryList;
    }
 
 
    csuOvalGeographicCoordinates() {
        return ([[[40.576179, -105.080773]]]);
    }
    coloradoGeographicBoundaries() {
        // northwest and southeast corners of the state of Colorado
        return L.latLngBounds(L.latLng(41, -109), L.latLng(37, -102));
    }
 
    createDistanceTable(){
 
        let entryList = this.createTableEntryList();
        return (
 
            <Table>
                <thead>
                <tr>
                    <th>#</th>
                    <th>Name</th>
                    <th>Latitude</th>
                    <th>Longitude</th>
                    <th>Distance To Next</th>
                    <th>Total Distance</th>
                </tr>
                </thead>
                <tbody>
                    {entryList.map( entryListItem =>
                        <tr>
                            <th scope="row">{entryListItem[0]}</th>
                            <td>{entryListItem[1]}</td>
                            <td>{entryListItem[2]}</td>
                            <td>{entryListItem[3]}</td>
                            <td>{entryListItem[4]} {this.props.options.activeUnit}</td>
                            <td>{entryListItem[5]} {this.props.options.activeUnit}</td>
                        </tr>
                    )}
                </tbody>
            </Table>
        );
    }

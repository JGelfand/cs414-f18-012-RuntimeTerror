import React, {Component} from 'react'
import {
    Row,
    Col, Button, ButtonGroup
} from 'reactstrap';
 
export default class ItineraryListItem extends Component{
    constructor(props) {
        super(props);
 
    }
 
    render() {
        return(
            <div>
                <Row>
                    <Col xs="6" sm="3">{this.props.itineraryListItem.name}</Col>
                    <Col xs="6" sm="3">{this.props.itineraryListItem.latitude}</Col>
                    <Col xs="6" sm="3">{this.props.itineraryListItem.longitude}</Col>
                    <Col xs="6" sm="3">
 
                        <ButtonGroup>
                            <Button onClick={this.moveLocation.bind(this, this.props.itineraryListItem, "up")}>↑</Button>
                            <Button onClick={this.moveLocation.bind(this, this.props.itineraryListItem, "down")}>↓</Button>
                            <Button onClick={this.removeLocation.bind(this, this.props.itineraryListItem)}>-</Button>
                            <Button onClick={this.moveLocation.bind(this, this.props.itineraryListItem, "top")}>Start</Button>
 
                        </ButtonGroup>
                    </Col>
                </Row>
            </div>
        )
    }
 
    onChange(){
 
    }
 
    removeLocation(itineraryListItem){
        this.props.removeLocation(itineraryListItem);
    }
    moveLocation(itineraryListItem, direction){
        this.props.moveLocation(itineraryListItem, direction);
    }
 
 
}
